  9      var charBuffer = function() {
 10          var buffer = '';
 11 
 12          return {
 13              length: buffer.length,
 14              get: function(len) {
 15                  var data, datalen;
 16 
 17                  len = len || 1;
 18                  datalen = (buffer.length > len) ? len : buffer.length;
 19                  data = buffer.slice(0, datalen);
 20                  buffer = buffer.slice(datalen);
 21                  this.length = buffer.length;
 22                  return data;
 23              },
 24              put: function(buf, len) {
 25                  buffer = buffer.concat(buf);
 26                  this.length = buffer.length;
 27              },
 28              clear: function() {
 29                  buffer = '';
 30                  this.length = buffer.length;
 31              }
 32          };
 33      };
 34 
 35      function makeSocket(scheme, type) {
 36          var sock, _jsSocket, rbuf, wbuf;
 37 
 38          function makeURL(args) {
 39              var host, port, path, url;
 40 
 41              if (args && args.host && args.host.length > 0) {
 42                  host = args.host;
 43              } else {
 44                  host = location.hostname;
 45              }
 46              if (host.match(/:/g)) {
 47                  host = host.match(/^\[.*\]$/) ? host : '[' + host + ']';
 48              }
 49              if (args && args.port) {
 50                  port = parseInt(args.port, 10);
 51              } else {
 52                  port = parseInt(location.port) ||
 53                      (scheme.match(/wss:/) ? 443 : 80);
 54              }
 55              path = (args && args.path) ? args.path : '';
 56              path = path.match(/^\//) ? path : '/' + path;
 57              if ((scheme.match(/ws:/) && port == 80) ||
 58                  (scheme.match(/wss:/) && port == 443)) {
 59                  url = scheme + '//' + host + path;
 60              } else {
 61                  url = scheme + '//' + host + ':' + port + path;
 62              }
 63              return url;
 64          }
 65 
 66          function onopen(e) {
 67              _jsSocket.onopen && _jsSocket.onopen(_jsSocket);
 68          }
 69 
 70          function onmessage(e) {
 71              rbuf.put(e.data);
 72              _jsSocket.onmessage && _jsSocket.onmessage(_jsSocket);
 73          }
 74 
 75          function onerror(e) {
 76              rbuf.clear();
 77              wbuf.clear();
 78              sock && sock.close();
 79              sock && delete sock;
 80              _jsSocket.onerror && _jsSocket.onerror(e);
 81          }
 82 
 83          function onclose(e) {
 84              rbuf.clear();
 85              wbuf.clear();
 86              sock && delete sock;
 87              _jsSocket.onclose && _jsSocket.onclose(e);
 88          }
 89 
 90          function doSend() {
 91              if (sock.bufferedAmount == 0) {
 92                  sock.send(wbuf.get(wbuf.length));
 93                  wbuf.clear();
 94              } else {
 95                  setTimeout(doSend, 50, false);
 96              }
 97          }
 98 
 99          if (type === 'binary') {
100              ; // make ArrayBuffer
101          } else {
102              rbuf = new charBuffer();
103              wbuf = new charBuffer();
104          }
105          /**
106           * base Class for Socket and SSL Socket
107           * @namespace
108           */
109          _jsSocket = {
110              /**
111               * send data
112               * @memberOf _jsSocket.prototype
113               * @param buf data to send
114               * @param [len] data length
115               */
116              send: function(buf, len) {
117                  wbuf.put(buf, len);
118                  doSend();
119              },
120              /**
121               * receive data
122               * @memberOf _jsSocket.prototype
123               * @param len data length
124               * @return data
125               */
126              recv: function(len) {
127                  var d = rbuf.get(len);
128 
129                  if (rbuf.length > 0) {
130                      setTimeout(function() {
131                                     _jsSocket.onmessage &&
132                                     _jsSocket.onmessage(_jsSocket);
133                                 }, 0);
134                  }
135                  return d;
136              },
137              /**
138               * connect to a server
139               * @memberOf _jsSocket.prototype
140               * @param {json} args
141               * @param {string} args.host FQDN or IPAddr
142               * @param {Number} [args.port] port number
143               * @param {string} args.path path
144               */
145              connect: function(args) {
146                  var url = makeURL(args);
147 
148                  try {
149                      sock = new WebSocket(url);
150                  } catch (x) {
151                      onerror(x);
152                      return;
153                  }
154                  sock.onopen = onopen;
155                  sock.onmessage = onmessage;
156                  sock.onerror = onerror;
157                  sock.onclose = onclose;
158              },
159              /**
160               * disconnect from a server
161               * @memberOf _jsSocket.prototype
162               */
163              disconnect: function() {
164                  sock && sock.close();
165              }
166              /**
167               * fire when connected a server
168               * @name _jsSocket#onopen
169               * @event
170               * @param {Event} e Event object
171               */
172              /**
173               * fire when received message from a server
174               * or messages exist in read queue
175               * @name _jsSocket#onmessage
176               * @event
177               * @param {MessageEvent} e MessageEvent object
178               */
179              /**
180               * fire when occured an error
181               * @name _jsSocket#onerror
182               * @event
183               * @param {Error} e Error object
184               */
185              /**
186               * fire when disconnected from a server
187               * @name _jsSocket#onclose
188               * @event
189               * @param {CloseEvent} e CloseEvent object
190               */
191          };
192          return _jsSocket;
193      }
194      /**
195       * Socket Class
196       * @constructor
197       * @augments _jsSocket
198       * @param {string} [type] 'text' or 'binary'<br>
199       * 'text' as default
200       */

210       * 'text' as default
211       */
212      jsSSLSocket = jsSSLSocket ||

214              return makeSocket('wss:', type);
215          };
