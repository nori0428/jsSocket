  9      var charBuffer = function() {
 10          var buffer = '';
 11 
 12          return {
 13              length: 0,
 14              get: function(len) {
 15                  var data, datalen;
 16 
 17                  if (this.length == 0 || len == 0) {
 18                      return undefined;
 19                  }
 20                  datalen = (buffer.length > len) ? len : buffer.length;
 21                  data = buffer.slice(0, datalen);
 22                  buffer = buffer.slice(datalen);
 23                  this.length = buffer.length;
 24                  return data;
 25              },
 26              put: function(buf) {
 27                  buffer = buffer.concat(buf);
 28                  this.length = buffer.length;
 29              },
 30              clear: function() {
 31                  buffer = '';
 32                  this.length = buffer.length;
 33              }
 34          };
 35      };
 36 
 37      var arrayBuffer = function() {
 38          var buffer;
 39 
 40          return {
 41              length: 0,
 42              get: function(len) {
 43                  var data, datalen;
 44 
 45                  if (!buffer || this.length == 0 || len == 0) {
 46                      return undefined;
 47                  }
 48                  datalen = (this.length > len) ? len : this.length;
 49                  data = buffer.subarray(0, datalen);
 50                  buffer = buffer.subarray(datalen);
 51                  this.length = buffer.length;
 52                  if (buffer.length == 0) {
 53                      buffer = undefined;
 54                  }
 55                  return data;
 56              },
 57              put: function(buf) {
 58                  var tmp, u8a = new Uint8Array(buf);
 59 
 60                  if (!buffer) {
 61                      buffer = u8a;
 62                  } else {
 63                      tmp = new Uint8Array(this.length + u8a.length);
 64                      tmp.set(buffer);
 65                      tmp.set(u8a, this.length);
 66                      buffer = tmp;
 67                  }
 68                  this.length = buffer.length;
 69              },
 70              clear: function() {
 71                  buffer = undefined;
 72                  this.length = 0;
 73              }
 74          };
 75      };

 77      function makeSocket(scheme, type) {
 78          var sock, _jsSocket, rbuf, wbuf, callbacked = false;
 79 
 80          function makeURL(args) {
 81              var host, port, path, url;
 82 
 83              if (args && args.host && args.host.length > 0) {
 84                  host = args.host;
 85              } else {
 86                  host = location.hostname;
 87              }
 88              if (host.match(/:/g)) {
 89                  host = host.match(/^\[.*\]$/) ? host : '[' + host + ']';
 90              }
 91              if (args && args.port) {
 92                  port = parseInt(args.port, 10);
 93              } else {
 94                  port = parseInt(location.port) ||
 95                      (scheme.match(/wss:/) ? 443 : 80);
 96              }
 97              path = (args && args.path) ? args.path : '';
 98              path = path.match(/^\//) ? path : '/' + path;
 99              if ((scheme.match(/ws:/) && port == 80) ||
100                  (scheme.match(/wss:/) && port == 443)) {
101                  url = scheme + '//' + host + path;
102              } else {
103                  url = scheme + '//' + host + ':' + port + path;
104              }
105              return url;
106          }

108          function onopen(e) {
109              _jsSocket.onopen && _jsSocket.onopen(_jsSocket);
110          }
111 
112          function onmessage(e) {
113              rbuf.put(e.data);
114              if (_jsSocket.onmessage && !callbacked) {
115                  callbacked = true;
116                  _jsSocket.onmessage(_jsSocket);
117              }
118          }
119 
120          function onerror(e) {
121              rbuf.clear();
122              wbuf.clear();
123              sock && sock.close();
124              sock && delete sock;
125              _jsSocket.onerror && _jsSocket.onerror(e);
126          }
127 
128          function onclose(e) {
129              rbuf.clear();
130              wbuf.clear();
131              sock && delete sock;
132              _jsSocket.onclose && _jsSocket.onclose(e);
133          }
134 
135          function doSend() {
136              var buf;
137 
138              if (sock.bufferedAmount == 0) {
139                  buf = wbuf.get(wbuf.length);
140                  if (!buf) {
141                      return;
142                  }
143                  if (typeof buf === 'string') {
144                      sock.send(buf);
145                  } else { // arrayBuffer
146                      sock.send(buf.buffer);
147                  }
148              } else {
149                  setTimeout(doSend, 50);
150              }
151          }
152 
153          if (type === 'binary') {
154              rbuf = new arrayBuffer();
155              wbuf = new arrayBuffer();
156          } else {
157              rbuf = new charBuffer();
158              wbuf = new charBuffer();
159          }
160          /**
161           * base Class for Socket and SSL Socket
162           * @namespace
163           */
164          _jsSocket = {
165              /**
166               * send data
167               * @memberOf _jsSocket.prototype
168               * @param buf data to send
169               */
170              send: function(buf) {
171                  wbuf.put(buf);
172                  doSend();
173              },
174              /**
175               * receive data
176               * @memberOf _jsSocket.prototype
177               * @param len data length
178               * @return {undefined || string || Uint8Array} data
179               * <p>
180               * returns 'string' when sock type === 'text'<br>
181               * returns 'Uint8Array' Object when sock type === 'binary'
182               */
183              recv: function(len) {
184                  var d;
185 
186                  d = rbuf.get(len);
187                  if (rbuf.length > 0) {
188                      setTimeout(function() {
189                                     _jsSocket.onmessage &&
190                                     _jsSocket.onmessage(_jsSocket);
191                                 }, 0);
192                  } else {
193                      callbacked = false;
194                  }
195                  return d;
196              },
197              /**
198               * connect to a server
199               * @memberOf _jsSocket.prototype
200               * @param {json} args
201               * @param {string} [args.host] FQDN or IPAddr<br>
202               * used location.hostname by default
203               * @param {Number} [args.port] port number<br>
204               * used location.port by default
205               * @param {string} args.path path
206               */
207              connect: function(args) {
208                  var url = makeURL(args);
209 
210                  try {
211                      if (window.WebSocket) {
212                          sock = new WebSocket(url);
213                      } else if (window.MozWebSocket) {
214                          sock = new MozWebSocket(url);
215                      }
216                  } catch (x) {
217                      onerror(x);
218                      return;
219                  }
220                  if (type === 'binary') {
221                      sock.binaryType = 'arraybuffer';
222                  }
223                  sock.onopen = onopen;
224                  sock.onmessage = onmessage;
225                  sock.onerror = onerror;
226                  sock.onclose = onclose;
227              },
228              /**
229               * disconnect from a server
230               * @memberOf _jsSocket.prototype
231               */
232              disconnect: function() {
233                  sock && sock.close();
234              }
235              /**
236               * fire when connected a server
237               * @name _jsSocket#onopen
238               * @event
239               * @param {Event} e Event object
240               */
241              /**
242               * fire when received message from a server
243               * or messages exist in read queue
244               * @name _jsSocket#onmessage
245               * @event
246               * @param {MessageEvent} e MessageEvent object
247               */
248              /**
249               * fire when occured an error
250               * @name _jsSocket#onerror
251               * @event
252               * @param {Error} e Error object
253               */
254              /**
255               * fire when disconnected from a server
256               * @name _jsSocket#onclose
257               * @event
258               * @param {CloseEvent} e CloseEvent object
259               */
260          };
261          return _jsSocket;
262      }

264       * Socket Class
265       * @constructor
266       * @augments _jsSocket
267       * @param {string} [type] 'text' or 'binary'<br>
268       * 'text' as default
269       */

276       * @constructor
277       * @augments _jsSocket

280       */
281      jsSSLSocket = jsSSLSocket ||

283              return makeSocket('wss:', type);
284          };
